# ELasticSearch

# 安装

yum install -y elasticsearch-6.8.12.rpm
yum install -y elasticsearch-7.8.1-x86_64.rpm

# 初始数据和日志目录

|---data
    |---app
    |---|---elasticsearch             
    |---|---|---data                        (数据目录)
    |---|---|---logs                        (日志目录)

# 新建数据目录和日志目录 
mkdir -p /data/app/data/elasticsearch/ 
mkdir -p /data/app/data/elasticsearch/data
mkdir -p /data/app/data/elasticsearch/logs
# 赋予目录权限
chown -R elasticsearch.elasticsearch /data/app/data/elasticsearch/
# 删除数据目录
rm -rf /data/app/data/elasticsearch
rm -rf /data/app/data/elasticsearch/data/*
rm -rf /data/app/data/elasticsearch/logs/*
# 删除默认数据目录
rm -rf /var/lib/elasticsearch
rm -rf /var/log/elasticsearch

# 维护

系统服务

# 随系统启动
systemctl enable elasticsearch.service
# 启动服务
systemctl start elasticsearch.service
# 停止服务
systemctl stop elasticsearch.service       
# 重启服务
systemctl restart elasticsearch.service
# 查看状态
systemctl status elasticsearch.service

查看安装目录

# 查看安装路径
whereis elasticsearch

验证是否启动成功

curl -XGET "127.0.0.1:9200"

修改配置文件

vi /etc/elasticsearch/elasticsearch.yml

参考配置项

# 允许其他机器访问，一般不建议全部开放
network.host: 0.0.0.0
# 数据目录
path.data : /data/app/data/elasticsearch/data
# 日志目录
path.logs : /data/app/data/elasticsearch/logs

# 插件

# elasticsearch-analysis-ik

中文分词

手工解压缩安装

# 6.8.6
