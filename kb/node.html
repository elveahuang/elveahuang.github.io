# NodeJS

# 常用命令

# Windows
npm config set prefix "D:\\Tools\\Node\\node-npm"
npm config set cache "D:\\Tools\\Node\\node-cache"
# MacOS
npm config set cache "/Users/elvea/Tools/node/node-cache"
npm config set prefix "/Users/elvea/Tools/node/node-npm"
# 设置镜像
npm config set registry="https://registry.npm.taobao.org"
npm config set registry="https://registry.npmjs.org"
# 设置代理
npm config set proxy http://127.0.0.1:7890
npm config set https-proxy http://127.0.0.1:7890
# Yarn
yarn config set cache-folder D:/Tools/Node/yarn-cache
yarn config set cache-folder /Users/elvea/Tools/node/yarn-cache
yarn config get registry
yarn config set registry https://registry.yarnpkg.com
yarn config set registry https://registry.npm.taobao.org
