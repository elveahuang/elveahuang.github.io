# MySQL - Master & Slave

# CentOS

# 安装目录说明

这里假设两台服务器的IP如下

  • 主服务器(192.168.0.104)
  • 从服务器(192.168.0.105)

两台服务器的MySQL的安装目录如下：

|---data
    |---app
    |   |---mysql             
    |   |   |---data                        (数据目录)
    |   |   |---logs                        (日志目录)
    |   |   |   |---mysql.log               (运行日志)
    |   |   |   |---mysql-slow.log          (慢查询日志)

分别执行下面命令创建对应的目录和文件，并把目录和文件的访问权限赋予给mysql用户。

# 新建数据目录和日志目录 
mkdir /data/app/data/mysql/ 
mkdir /data/app/data/mysql/data
mkdir /data/app/data/mysql/logs
# 创建日志文件，这一步非必须
touch /data/app/data/mysql/logs/mysql.log
touch /data/app/data/mysql/logs/mysql-slow.log
# 赋予目录权限
chown -R mysql.mysql /data/app/data/mysql/
# 新建数据库目录并赋予权限，否则会报错导致服务无法成功启动
mkdir /var/lib/mysql
# 赋予目录权限
chown -R mysql.mysql /var/lib/mysql
# 
# 删除数据目录
rm -rf /data/app/data/mysql
rm -rf /data/app/data/mysql/data/*
rm -rf /data/app/data/mysql/logs/*
rm -rf /var/lib/mysql

# 安装和配置

安装可以参考MySQL的安装和配置，安装完成后修改配置文件后再启动。

主库最终配置文件如下：

# MySQL Master
[mysqld]
server-id=1
#
log_bin=/data/app/data/mysql/mysql-master-bin
log_bin_index=/data/app/data/mysql/mysql-master-bin.index
#
# 数据目录
datadir=/data/app/data/mysql/data
# 错误日志
log-error=/data/app/data/mysql/logs/mysql.log
# 最大文件打开数
open_files_limit=1024
# 使用旧版的认证，兼容传统客户端
default-authentication-plugin=mysql_native_password
# 表名不区分大小写
lower_case_table_names=1
# 默认编码
character-set-client-handshake=FALSE
character-set-server=utf8mb4
collation-server=utf8mb4_unicode_ci
# 指定时区
default-time-zone='+8:00'
# InnoDB
innodb_buffer_pool_size=8G
innodb_log_file_size=2G
innodb_log_buffer_size=4M
#
join_buffer_size=4M
#
sort_buffer_size=4M
#
read_buffer_size=8M
#
read_rnd_buffer_size=16M
#
key_buffer_size=128M
#
max_allowed_packet=16M
#
tmp_table_size=128M
#
table_open_cache=1024
# 最大连接数
max_connections=2500
# 慢查询
# 开启慢查询日志
slow-query-log=on
# 慢查询日志路径
slow_query_log_file=/data/app/data/mysql/logs/mysql-slow.log
# 修改为记录5秒内的查询，默认不设置此参数为记录10秒内的查询
long_query_time=5
# 记录未使用索引的查询
log-queries-not-using-indexes=on

[client]
default-character-set=utf8mb4

[mysql]
default-character-set=utf8mb4

[mysqldump]
quick
max_allowed_packet=8M

数据库安装成功后，从日志文件中查找临时密码，获取临时密码后，可以登录数据库并重置账号密码。

# 查找临时密码。
grep 'temporary password' /data/app/data/mysql/logs/mysql.log
# 用临时密码登录后，修改密码。
ALTER USER 'root'@'localhost' IDENTIFIED WITH mysql_native_password BY 'Root@007';
# 创建局域网内账号
CREATE USER 'root'@'192.168.%' IDENTIFIED BY 'Root@007';
GRANT ALL PRIVILEGES ON *.* TO 'root'@'192.168.%' WITH GRANT OPTION;

# 主从热备

# 主服务器

# 1. 查看主库信息

show master status\G;

# 从服务器

# 1. 连接到主库

change master to master_host='192.168.0.104',master_port=3306,master_user='root',master_password='Root@007',master_log_file='mysql-master-bin.000002',master_log_pos=979;

# 2. 启用同步

start slave;

# 3. 查看从库信息

show slave status\G;

# 4. 停止同步

stop slave;

# 5. 重置所有主库连接

reset slave all;
