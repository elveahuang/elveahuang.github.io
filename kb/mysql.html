# MySQL

# CentOS - 安装

# 1. 删除系统已安装包

当前大部分的发行版都预装了mariadb或mysql, 安装前需要检查并移除已安装的包。

# 删除预装的mysql相关包，如果存在的话。
rpm -qa | grep mysql
rpm -e --nodeps mysql-libs-5.1.73-8.el6_8.x86_64
rpm -qa | grep mysql
# 删除预装的mariadb相关包，如果存在的话。
rpm -qa | grep mariadb
rpm -e --nodeps mariadb-libs-5.5.65-1.el7.x86_64
rpm -qa | grep mariadb

# 2. 安装

# ----------------------------------------------------------------------------
# MySQL 8.0.x on CentOS 7.x 
# ---------------------------------------------------------------------------- 
# 解压缩
tar xvf mysql-8.0.21-1.el7.x86_64.rpm-bundle.tar
# 安装
yum install mysql-community-common-8.0.21-1.el7.x86_64.rpm
yum install mysql-community-libs-8.0.21-1.el7.x86_64.rpm
yum install mysql-community-libs-compat-8.0.21-1.el7.x86_64.rpm
yum install mysql-community-client-8.0.21-1.el7.x86_64.rpm
yum install mysql-community-server-8.0.21-1.el7.x86_64.rpm
# ----------------------------------------------------------------------------
# MySQL 8.0.x on CentOS 8.x
# ----------------------------------------------------------------------------
# 解压缩
tar xvf mysql-8.0.21-1.el8.x86_64.rpm-bundle.tar
# 安装
yum install mysql-community-common-8.0.19-1.el8.x86_64.rpm
yum install mysql-community-libs-8.0.19-1.el8.x86_64.rpm
yum install mysql-community-client-8.0.19-1.el8.x86_64.rpm
yum install mysql-community-server-8.0.19-1.el8.x86_64.rpm

# MySQL 5.7.x on CentOS 7.x
# ----------------------------------------------------------------------------

# 安装

yum install mysql-community-libs-5.7.29-1.el7.x86_64.rpm

# 3. 初始密码

数据库启动初始化完成后，会生成一个临时的密码保存在/var/log/mysqld.log的日志文件里面。 所以第一次登录时需要查到这个临时密码，并使用临时密码登录后修改初始密码。

# 查找临时密码。
grep 'temporary password' /var/log/mysqld.log
# 用临时密码登录后，修改密码。
ALTER USER 'root'@'localhost' IDENTIFIED WITH mysql_native_password BY 'Root@007';

# 4. 系统服务

# 随系统启动
systemctl enable mysqld.service
# 禁用服务
systemctl disable mysqld.service
# 启动服务
systemctl start mysqld.service
# 停止服务
systemctl stop mysqld.service
# 重启服务
systemctl restart mysqld.service
# 查看状态
systemctl status mysqld.service

# 5. 卸载

# 查询已有安装包
rpm -qa|grep mysql
# 删除已有安装包
# 8.0.x
rpm -e --nodeps mysql-community-common-8.0.19-1.el8.x86_64
rpm -e --nodeps mysql-community-client-8.0.19-1.el8.x86_64
rpm -e --nodeps mysql-community-server-8.0.19-1.el8.x86_64
rpm -e --nodeps mysql-community-libs-8.0.19-1.el8.x86_64
# 5.7.x
rpm -e --nodeps mysql-community-libs-compat-5.7.29-1.el7.x86_64
rpm -e --nodeps mysql-community-common-5.7.29-1.el7.x86_64
rpm -e --nodeps mysql-community-server-5.7.29-1.el7.x86_64
rpm -e --nodeps mysql-community-client-5.7.29-1.el7.x86_64
rpm -e --nodeps mysql-community-libs-5.7.29-1.el7.x86_64
# 查询MySQL相关目录
find / -name mysql 
# 删除相关目录
rm -rf /var/lib/mysql
rm -rf /var/lib/mysql/mysql
rm -rf /usr/share/mysql
rm -rf /etc/selinux/targeted/active/modules/100/mysql
rm -rf /etc/selinux/targeted/tmp/modules/100/mysql

# Windows - 安装

最简单的配置文件

[mysqld]
#
character-set-server=utf8mb4
#
basedir=D:\Dev\Database\MySQL\mysql
datadir=D:\Dev\Database\MySQL\data
[client]
#
default-character-set = utf8mb4
[mysql]
#
default-character-set=utf8mb4

操作命令

# 初始化数据库
mysqld --initialize-insecure --user=root --datadir=D:/Tools/MySQL/Data
# 安装系统服务
mysqld --install MySQLServer --defaults-file=D:/Tools/MySQL/Config/my.ini
# 移除系统服务
mysqld –remove MySQLServer
# 启动
net start MySQLServer
# 停止
net stop MySQLServer
