    # MySQL - 常见问题

    # 安装后修改密码和新建账号

    # ====================================================================================================
# 新密码
# ====================================================================================================
# 修改密码
ALTER USER 'root'@'localhost' IDENTIFIED BY 'Root@007';
# 创建用户
CREATE USER 'root'@'192.168.%' IDENTIFIED BY 'Root@007';
# ====================================================================================================
# 传统密码
# ====================================================================================================
# 修改密码
ALTER USER 'root'@'localhost' IDENTIFIED WITH mysql_native_password BY 'Root@007';
# 创建用户
CREATE USER 'root'@'192.168.%' IDENTIFIED WITH mysql_native_password BY 'Root@007';
# ====================================================================================================
# 授权
GRANT ALL PRIVILEGES ON *.* TO 'root'@'192.168.%' WITH GRANT OPTION;
# 刷新权限
flush privileges;

    # 编码和字符集

    # 创建指定编码和字符集的数据库

CREATE DATABASE `test`
    DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci;

# 创建指定编码和字符集的表

create table test
(
    id int
) character set utf8mb4
  collate utf8mb4_general_ci;

# 查看表信息，包含列的字符集

SHOW FULL COLUMNS FROM sys_user;

# 修改表指定列的编码和字符集

ALTER TABLE test
    CHANGE id id VARCHAR(100) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci;

show master status;

purge binary logs to 'mysql-bin.000005';
