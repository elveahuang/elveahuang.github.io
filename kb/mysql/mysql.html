# MySQL - 安装

# CentOS - 安装

# 1. 删除系统已安装包

当前大部分的发行版都预装了mariadb或mysql, 安装前需要检查并移除已安装的包。

# 删除预装的mysql相关包，如果存在的话。
rpm -qa | grep mysql
rpm -e --nodeps mysql-libs-5.1.73-8.el6_8.x86_64
rpm -qa | grep mysql
# 删除预装的mariadb相关包，如果存在的话。
rpm -qa | grep mariadb
rpm -e --nodeps mariadb-libs-5.5.65-1.el7.x86_64
rpm -qa | grep mariadb

# 2. 安装

# ----------------------------------------------------------------------------
# CentOS 7.x
# ----------------------------------------------------------------------------
# 解压缩
tar xvf mysql-8.0.24-1.el7.x86_64.rpm-bundle.tar
# 安装
yum install mysql-community-common-8.0.24-1.el7.x86_64.rpm
yum install mysql-community-libs-8.0.24-1.el7.x86_64.rpm
yum install mysql-community-libs-compat-8.0.24-1.el7.x86_64.rpm
yum install mysql-community-client-8.0.24-1.el7.x86_64.rpm
yum install mysql-community-server-8.0.24-1.el7.x86_64.rpm
# ----------------------------------------------------------------------------
# CentOS 8.x
# ----------------------------------------------------------------------------
# 解压缩
tar xvf mysql-8.0.24-1.el8.x86_64.rpm-bundle.tar
# 安装
yum install mysql-community-common-8.0.24-1.el8.x86_64.rpm
yum install mysql-community-client-plugins-8.0.24-1.el8.x86_64.rpm
yum install mysql-community-libs-8.0.24-1.el8.x86_64.rpm
yum install mysql-community-client-8.0.24-1.el8.x86_64.rpm
yum install mysql-community-server-8.0.24-1.el8.x86_64.rpm

# 3. 初始密码

数据库启动初始化完成后，会生成一个临时的密码保存在/var/log/mysqld.log的日志文件里面。 所以第一次登录时需要查到这个临时密码，并使用临时密码登录后修改初始密码。

# 查找临时密码。
grep 'temporary password' /var/log/mysqld.log

获取到初始密码后可以通过修改密码来修改初始密码。

# 4. 系统服务

# 随系统启动
systemctl enable mysqld.service
# 禁用服务
systemctl disable mysqld.service
# 启动服务
systemctl start mysqld.service
# 停止服务
systemctl stop mysqld.service
# 重启服务
systemctl restart mysqld.service
# 查看状态
systemctl status mysqld.service

# 5. 卸载

# 查询已有安装包
rpm -qa|grep mysql
# 删除已有安装包
# 8.0.x
rpm -e --nodeps mysql-community-common-8.0.24-1.el8.x86_64
rpm -e --nodeps mysql-community-client-8.0.24-1.el8.x86_64
rpm -e --nodeps mysql-community-server-8.0.24-1.el8.x86_64
rpm -e --nodeps mysql-community-libs-8.0.24-1.el8.x86_64
# 查询MySQL相关目录
find / -name mysql 
# 删除相关目录
rm -rf /var/lib/mysql
rm -rf /var/lib/mysql/mysql
rm -rf /usr/share/mysql
rm -rf /etc/selinux/targeted/active/modules/100/mysql
rm -rf /etc/selinux/targeted/tmp/modules/100/mysql

# MacOS - 安装

# 卸载

sudo rm /usr/local/mysql
sudo rm -rf /usr/local/mysql*
sudo rm -rf /Library/StartupItems/MySQLCOM
sudo rm -rf /Library/PreferencePanes/My*
sudo rm -rf ~/Library/PreferencePanes/My*
sudo rm -rf /Library/Receipts/mysql*
sudo rm -rf /Library/Receipts/MySQL*
sudo rm -rf /var/db/receipts/com.mysql.*

# Windows - 安装

安装目录说明

|---D
    |---Tools
    |---|---MySQL             
    |---|---|---Server                          (安装目录)
    |---|---|---Data                            (数据目录)
    |---|---|---Logs                            (日志目录)
    |---|---|---|---mysql.log                   (运行日志)
    |---|---|---|---mysql-slow.log              (慢查询日志)
    |---|---|---my.ini                          (配置文件)

操作命令

# 初始化数据库
mysqld --initialize-insecure --datadir=D:\Tools\MySQL\Data
# 安装系统服务
mysqld --install MySQL --defaults-file=D:\Tools\MySQL\my.ini
# 手工启动服务
mysqld --defaults-file=D:\Tools\MySQL\my.ini
# 移除系统服务
mysqld –remove MySQL
# 启动
net start MySQL
# 停止
net stop MySQL

参考配置文件

[mysqld]
# 端口
port = 3306
# 安装目录和数据目录
basedir = D:\Tools\MySQL\Server
datadir = D:\Tools\MySQL\Data
# 表名不区分大小写
lower_case_table_names = 1
# 编码
character-set-server = utf8mb4
# 字符集
collation-server = utf8mb4_0900_ai_ci
#
init_connect = 'SET NAMES utf8mb4'
# 指定时区
default-time-zone = '+8:00'
# 启用传统密码策略认证
default_authentication_plugin = mysql_native_password
# 错误日志
log-error = D:\Tools\MySQL\Logs\mysql.log
# 慢查询
# 开启慢查询日志
slow-query-log = on
# 慢查询日志路径
slow_query_log_file = D:\Tools\MySQL\Logs\mysql-slow.log
# 修改为记录5秒内的查询，默认为记录10秒内的查询
long_query_time = 5
# 记录未使用索引的查询
log-queries-not-using-indexes = on
[client]
default-character-set = utf8mb4
[mysql]
default-character-set = utf8mb4
