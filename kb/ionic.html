# Android

# 证书

# ****************************************************************************************************************
# JKS
# **************************************************************************************************************** 
# 生成秘钥
keytool -genkey -alias android -keyalg RSA -validity 36500 -keysize 2048 -keystore android.jks -keypass android -storepass android 
# JKS -> PKCS12
keytool -importkeystore -srckeystore android.jks -destkeystore android.jks -deststoretype pkcs12
# 生成公钥 
keytool -export -alias android -keystore android.jks -storepass android -rfc -file android.cer
#
# ****************************************************************************************************************
# PKCS12
# ****************************************************************************************************************
# 生成秘钥
keytool -genkey -alias android -keyalg RSA -storetype PKCS12 -validity 36500 -keysize 2048 -keystore android.jks -keypass android -storepass android
# 生成公钥
keytool -export -alias android -keystore android.jks -storetype PKCS12 -storepass android -rfc -file android.cer
#
# ****************************************************************************************************************
# 查看证书
# ****************************************************************************************************************
keytool -list -keystore android.jks -storepass android
keytool -list -v -keystore android.jks -storepass android
keytool -list -rfc -keystore android.jks -storepass android

# 编译打包

# 
# 真机运行
# 
ionic cordova run android --device --buildConfig=build-android.json --target xxxxxx
#
# 调试包
ionic cordova build android --prod --debug --device --buildConfig=build-android.json
# 生产包
ionic cordova build android --prod --release --device --buildConfig=build-android.json
