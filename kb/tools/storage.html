RustFS ​

RHEL ​

安装

bash
# 安装目录
mkdir -p /data/app/tools/rustfs
# 下载
wget https://dl.rustfs.com/artifacts/rustfs/release/rustfs-linux-x86_64-musl-latest.zip
unzip rustfs-linux-x86_64-musl-latest.zip -d /data/app/tools/rustfs
chmod +x /data/app/tools/rustfs/rustfs
# 数据目录
sudo mkdir -p /data/app/data/rustfs/data /data/app/data/rustfs/logs /data/app/data/rustfs/tls
sudo chmod -R 750 /data/app/data/rustfs/data /data/app/data/rustfs/logs /data/app/data/rustfs/tls

启动脚本

bash
#!/bin/sh
########################################################################################################################
# Minio
########################################################################################################################
export RUSTFS_HOME=/data/app/tools/rustfs
export RUSTFS_DATA_HOME=/data/app/data/rustfs/data
export RUSTFS_LOGS_HOME=/data/app/data/rustfs/logs
#
export RUSTFS_ADDRESS=0.0.0.0:9000
export RUSTFS_ACCESS_KEY=admin
export RUSTFS_SECRET_KEY=admin
export RUSTFS_CONSOLE_ENABLE=true
export RUSTFS_SERVER_DOMAINS=s.wdev.cn
export RUST_LOG=info
export RUSTFS_VOLUMES=/data/rustfs/data
export RUSTFS_OBS_LOG_DIRECTORY=/data/rustfs/logs
#
echo "Current RustFS Home - $RUSTFS_HOME"
echo "Current RustFS Data - $RUSTFS_VOLUMES"
# Start RustFS Server
echo Start Minio Server......
nohup $RUSTFS_HOME/rustfs > $RUSTFS_LOGS_HOME/rustfs.log 2>&1 &

bash
#!/bin/sh
########################################################################################################################
# Minio
########################################################################################################################
export MINIO_HOME=/data/app/tools/minio
export MINIO_PORT=9000
export MINIO_CONSOLE_PORT=9001
export MINIO_DATA_HOME=/data/app/data/minio
export MINIO_ROOT_USER=admin
export MINIO_ROOT_PASSWORD=Root@123456
export PATH="$MINIO_HOME/bin:$PATH"
#
echo "Current Minio Home - $MINIO_HOME"
echo "Current Minio Data Home - $MINIO_DATA_HOME"
# Start Minio Server
echo Start Minio Server......
nohup $MINIO_HOME/minio server $MINIO_DATA_HOME --address ":$MINIO_PORT" --console-address ":$MINIO_CONSOLE_PORT" > $MINIO_HOME/minio.log 2>&1 &
