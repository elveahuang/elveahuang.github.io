# OnlyOffice Document Server on CentOS

  • OnlyOffice (opens new window)

  • OnlyOffice Github (opens new window)

#
#
# 1. Only Office Document Server
docker pull onlyoffice/documentserver
# 
docker run -i -t -d --name onlyoffice -p 9880:80 --network host onlyoffice/documentserver
#
docker run -i -t -d --name onlyoffice -p 0.0.0.0:9880:80 -p 0.0.0.0:9881:443 \
--restart always onlyoffice/documentserver
# 查看容器
docker ps
# 进入容器
docker exec -u root -it 89c59ab2cc97 /bin/bash
# 停止容器 
docker stop 89c59ab2cc97
# 重启容器
docker restart 89c59ab2cc97
# 删除容器 
docker rm 89c59ab2cc97
#
#
mkdir -p /data/app/tools/onlyoffice/ssl
mkdir -p /data/app/tools/onlyoffice/logs
mkdir -p /data/app/tools/onlyoffice/data
# SSL
# 1. 创建私钥
openssl genrsa -out onlyoffice.key 2048
# 2. 创建CSR
openssl req -new -key onlyoffice.key -out onlyoffice.csr
# 3. 用私枂和CSR签发证书
openssl x509 -req -days 365 -in onlyoffice.csr -signkey onlyoffice.key -out onlyoffice.crt
# 4. 用dhparam加密服务器密钥
openssl dhparam -out dhparam.pem 2048

#
docker run -i -t -d --name onlyoffice -p 9880:80 \ 
-v /data/app/tools/onlyoffice/logs:/var/log/onlyoffice \ 
-v /data/app/tools/onlyoffice/data:/var/www/onlyoffice/Data \ 
--restart=always onlyoffice/documentserver
