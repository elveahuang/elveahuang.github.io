# LibreOffice on CentOS

# 解压缩安装包

tar zxvf LibreOffice_7.0.0_Linux_x86-64_rpm.tar.gz
tar zxvf LibreOffice_7.0.0_Linux_x86-64_rpm_langpack_zh-CN.tar.gz

# 全量安装

yum install -y ./LibreOffice_7.0.0.3_Linux_x86-64_rpm/RPMS/*.rpm
yum install -y ./LibreOffice_7.0.0.3_Linux_x86-64_rpm_langpack_zh-CN/RPMS/*.rpm

# 查看并安装缺少依赖包

查看缺失依赖包

ldd /opt/libreoffice7.0/program/soffice.bin

安装缺少的依赖项

yum install -y cairo
yum install -y libSM
yum install -y ibus
yum install -y libX11-xcb

# 新建链接

ln -s /opt/libreoffice7.0 /opt/libreoffice

# 卸载

yum erase libreoffice\*
