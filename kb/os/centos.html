如果buff/cache占用太高，可以通过下面命令清空缓存

# 清除pagecache
echo 1 > /proc/sys/vm/drop_caches
# 清除回收slab分配器中的对象
echo 2 > /proc/sys/vm/drop_caches
# 清除pagecache和slab分配器中的缓存对象
echo 3 > /proc/sys/vm/drop_caches

# 编辑配置文件

/etc/exports

配置文件参考如下

/data/app/webapp/www/resources 192.168.0.102(rw,sync,no_root_squash)
/data/app/webapp/www/resources 192.168.0.103(rw,sync,no_root_squash)
#
/data/app/webapp/www/resources 192.168.0.*(rw,sync,no_root_squash)
  • rw表示可读写
  • ro表示只读
  • sync表示将数据同步写入内存缓冲区与磁盘中，效率低，但可以保证数据的一致性
  • no_root_squash表示当访问者为root用户时访问该目录具有root权限
echo /data/app/webapp/www/resources 192.168.0.102(rw,sync,no_root_squash) >> /etc/exports
echo /data/app/webapp/www/resources 192.168.0.102(rw,sync,no_root_squash) >> /etc/exports
echo /data/app/webapp/www/resources 192.168.0.* (rw,sync,no_root_squash) >> /etc/exports

# 防火墙

除了2049和111两个端口固定以外，其他端口都是随机的，最好的办法是自定义对应端口，以便配置防火墙。

# CentOS 7.x
vi /etc/sysconfig/nfs
# 追加端口配置
RQUOTAD_PORT=875
LOCKD_TCPPORT=32803
LOCKD_UDPPORT=32769
MOUNTD_PORT=892
STATD_PORT=662
# CentOS 8.x
vi /etc/nfs.conf
# 修改一下端口配置
[lockd]
port = 32803
udp-port = 32769
...
[mountd]
port = 892
...
[statd]
port = 662

开放相关端口

# 手工挂载和取消挂载

挂载

mount -o tcp,nolock 192.168.0.101:/data/app/webapp/www/resources /data/app/webapp/www/resources
mount -t nfs 192.168.0.101:/data/app/webapp/www/resources /data/app/webapp/www/resources
  • nfs表示使用的协议
  • nolock表示非阻塞

取消挂载

umount 192.168.0.101:/data/app/webapp/www/resources

# 自动挂载

# 往fstab文件添加自动挂载配置
echo "192.168.0.101:/data/app/webapp/www/resources  /data/app/webapp/www/resources  nfs  defaults  0  0"  >>  /etc/fstab
# 人工编辑fatab文件添加自动挂载内容

打开fstab

vi /etc/fstab

添加一下配置

192.168.0.101:/data/app/webapp/www/resources  /data/app/webapp/www/resources  nfs  defaults  0  0

使配置生效

mount -a
