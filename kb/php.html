# XDebug & MacOS

# 编译安装

#
brew install autoconf
#
sudo xcode-select -switch /Applications/Xcode.app
#
/Applications/XAMPP/xamppfiles/bin/phpize-7.4.1
#
./configure --enable-xdebug --with-php-config=/Applications/XAMPP/xamppfiles/bin/php-config-7.4.1
#
make
#
sudo cp modules/* /Applications/XAMPP/xamppfiles/lib/php/extensions/no-debug-non-zts-20190902/

# 配置

[XDebug]
zend_extension = "/Applications/XAMPP/xamppfiles/lib/php/extensions/no-debug-non-zts-20170718/xdebug.so"
xdebug.idekey = "XDebugIDE"
xdebug.remote_enable = 1
xdebug.remote_host = "127.0.0.1"
xdebug.remote_port = 9001
xdebug.profiler_enable = 1
xdebug.profiler_output_dir = "/Users/elvea/Work/temp"
