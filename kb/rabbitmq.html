# RabbitMQ

# 安装 - CentOS

# 1. ErLang

RabbitMQ依赖于ErLang，安装前置条件就是安装好ErLang (opens new window)，RabbitMQ提供了CentOS的RPM包，可以从下面地址下载 https://github.com/rabbitmq/erlang-rpm/releases (opens new window)。

# 安装ErLang
# CentOS 7.x
yum install -y erlang-23.0.3-1.el7.x86_64.rpm
# CentOS 8.x
yum install -y erlang-23.0.2-1.el8.x86_64.rpm

# 2. 安装

# CentOS 7.x
yum install -y rabbitmq-server-3.8.8-1.el7.noarch.rpm
# CentOS 8.x
yum install -y rabbitmq-server-3.8.8-1.el8.noarch.rpm

# 3. 维护

系统服务

# 开机启动    
systemctl enable rabbitmq-server.service
# 禁用开机启动    
systemctl disable rabbitmq-server.service
# 启动服务  
systemctl start rabbitmq-server.service
# 重启服务  
sy
# 关闭服务    
systemctl stop rabbitmq-server.service
# 查看状态
systemctl status rabbitmq-server.service

检查

# 查看安装路径
whereis rabbitmq
# 检查端口
netstat -lanp | grep 5672
netstat -lanp | grep 15672
# 查看进程
ps -ef | grep rabbitmq

启用相关管理插件

rabbitmq-plugins enable rabbitmq_management

默认RabbitMQ的guest账号只允许本地登录，所以需要添加新的管理员账号，这样就可以从外部登录页面管理后台。

[root@localhost rabbitmq]# rabbitmqctl add_user admin admin
Creating user "admin" ...
[root@localhost rabbitmq]# rabbitmqctl set_permissions -p "/" admin ".*" ".*" ".*"
Setting permissions for user "admin" in vhost "/" ...
[root@localhost rabbitmq]# rabbitmqctl  set_user_tags admin administrator
Setting tags for user "admin" to [administrator] ...

# 安装 - MacOS

安装

tar -Jxvf rabbitmq-server-generic-unix-3.8.9.tar.xz
# 解压缩
mv rabbitmq_server-3.8.9/* /Users/elvea/Tools/rabbitmq
# 删除源代码目录
rm -rf rabbitmq_server-3.8.9
rm -rf rabbitmq-server-generic-unix-3.8.9.tar.xz

启用管理插件

cd /Users/elvea/Tools/rabbitmq/sbin
./rabbitmq-plugins enable rabbitmq_management

添加用户

## 添加用户
[root@localhost rabbitmq]# ./rabbitmqctl add_user admin admin
Creating user "admin" ...
[root@localhost rabbitmq]# ./rabbitmqctl set_permissions -p "/" admin ".*" ".*" ".*"
Setting permissions for user "admin" in vhost "/" ...
[root@localhost rabbitmq]# ./rabbitmqctl  set_user_tags admin administrator
Setting tags for user "admin" to [administrator] ...
[root@localhost rabbitmq]#

基本操作

# 查看队列
rabbitmqctl list_queues
# 关闭应用
rabbitmqctl stop_app
# 清空队列
rabbitmqctl reset
# 重启应用
rabbitmqctl start_app

# 2. 随机启动

新增加配置文件

sudo vim ~/Library/LaunchAgents/rabbitmq.plist

配置文件内容如下

<?xml version='1.0' encoding='UTF-8'?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version='1.0'>
    <dict>
        <key>Label</key>
        <string>rabbitmq</string>
        <key>ProgramArguments</key>
        <array>
            <string>/Users/elvea/Tools/rabbitmq/sbin/rabbitmq-server</string>
        </array>
        <key>RunAtLoad</key>
        <true/>
        <key>KeepAlive</key>
        <true/>
        <key>EnvironmentVariables</key>
        <dict>
            <key>PATH</key>
            <string>/usr/local/sbin:/usr/bin:/bin:/usr/local/bin</string>
        </dict>
        <key>WorkingDirectory</key>
        <string>/Users/elvea/Tools/rabbitmq/</string>
        <key>StandardErrorPath</key>
        <string>/Users/elvea/Tools/rabbitmq/logs/output.log</string>
        <key>StandardOutPath</key>
        <string>/Users/elvea/Tools/rabbitmq/logs/output.log</string>
        <key>HardResourceLimits</key>
        <dict>
            <key>NumberOfFiles</key>
            <integer>10240</integer>
        </dict>
        <key>SoftResourceLimits</key>
        <dict>
            <key>NumberOfFiles</key>
            <integer>10240</integer>
        </dict>
    </dict>
</plist>

操作

# 验证配置文件
plutil -lint ~/Library/LaunchAgents/rabbitmq.plist
# 添加开机启动并立刻启动
launchctl load -w ~/Library/LaunchAgents/rabbitmq.plist
# 取消开机启动并立刻停止 
launchctl unload -w ~/Library/LaunchAgents/rabbitmq.plist

# 安装 - Windows

# 1. 安装ErLang

当前ErLang的最新版本是的官方下载链接是:

Win 32 (opens new window)和Win 64 (opens new window)

推荐使用64版本安装，最简单的方式是直接按默认的直接安装，安装完成后，设置好环境变量ERLANG_HOME，并把bin目录加进PATH里面。

ERLANG_HOME=D:\Dev\Erlang\ERL

PATH=%ERLANG_HOME%\bin;%PATH%

# 2. 安装RabbitMQ

安装详细介绍可以查看官方文档，Manual (opens new window) 和 With installer (opens new window)

下载安装使用绿色安装做例子。

RabbitMQ最新版的下载链接是rabbitmq-server-windows-3.7.0.zip (opens new window)

下载完成后，设置好环境变量并把sbin目录加进PATH里面，方便直接命令行里面操作RabbitMQ。

比如

RABBITMQ_HOME=D:\Java\rabbitmq

PATH=%RABBITMQ_HOME%\sbin;%PATH%

# 3. 配置RabbitMQ

安装系统服务

rabbitmq-service.bat install

启用管理插件

rabbitmq-plugins.bat enable rabbitmq_management

启用服务

rabbitmq-service.bat start

移除系统服务

rabbitmq-service.bat remove

浏览器访问http://localhost:15672 (opens new window) 默认账号: guest 密码: guest。

管理中间件出现错误的可能解决办法

错误大概如下：

Error: unable to connect to node 'rabbit@xxxx-pc': nodedown

DIAGNOSTICS
===========

attempted to contact: ['rabbit@elvea-pc']

rabbit@elvea-pc:
  * connected to epmd (port 4369) on xxxx-pc
  * epmd reports node 'rabbit' running on port 25672
  * TCP connection succeeded but Erlang distribution failed

  * Authentication failed (rejected by the remote node), please check the Erlang cookie

因为ErLang的cookie文件没有和RabbitMQ的cokkie同步引起的，所以只需要复制C:\windows.erlang.cookie文件替换用户目录下的（如：C:\users\xxx）.erlang.cookie文件就可以。
