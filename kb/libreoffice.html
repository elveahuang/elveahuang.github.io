# LibreOffice

# CentOS

# 解压缩安装包

tar zxvf LibreOffice_6.4.2_Linux_x86-64_rpm.tar.gz
tar zxvf LibreOffice_6.4.2_Linux_x86-64_rpm_langpack_zh-CN.tar.gz

# 全量安装

yum install ./LibreOffice_6.4.2.2_Linux_x86-64_rpm/RPMS/*.rpm
yum install ./LibreOffice_6.4.2.2_Linux_x86-64_rpm_langpack_zh-CN/RPMS/*.rpm

# 查看并安装缺少依赖包

查看缺失依赖包

ldd /opt/libreoffice6.4/program/soffice.bin

安装缺少的依赖项

yum install cairo
yum install libSM
yum install ibus

# 新建链接

ln -s /opt/libreoffice6.4 /opt/libreoffice

# 卸载

yum erase libreoffice\*
