RHEL - 在线安装 ​

镜像源

bash
# 安装官方镜像源
sudo yum install -y https://download.postgresql.org/pub/repos/yum/reporpms/EL-7-x86_64/pgdg-redhat-repo-latest.noarch.rpm
sudo yum install -y https://download.postgresql.org/pub/repos/yum/reporpms/EL-8-x86_64/pgdg-redhat-repo-latest.noarch.rpm
sudo yum install -y https://download.postgresql.org/pub/repos/yum/reporpms/EL-9-x86_64/pgdg-redhat-repo-latest.noarch.rpm
sudo yum install -y https://download.postgresql.org/pub/repos/yum/reporpms/EL-10-x86_64/pgdg-redhat-repo-latest.noarch.rpm
# 卸载镜像源
yum -y remove pgdg-redhat-repo-latest
# 替换国内镜 - 腾讯云
rpm -Uvh https://mirrors.tencent.com/postgresql/repos/yum/reporpms/EL-7-x86_64/pgdg-redhat-repo-latest.noarch.rpm
sed -i "s@https://download.postgresql.org/pub@https://mirrors.tencent.com/postgresql@g" /etc/yum.repos.d/pgdg-redhat-all.repo

MacOS - 在线安装 ​

bash
# 安装
# 安装后，把/opt/homebrew/opt/postgresql@18/bin路径加入环境变量
brew install postgresql@18
# 新版本安装时已经包含初始化，所以下面初始化无需执行
initdb --locale=C -E UTF-8 /opt/homebrew/var/postgresql@18
# 启动服务
brew services start postgresql@18
# 创建用户
createuser root -P
# 创建测试数据库
createdb test -O root -E UTF8 -e
# 链接测试数据库
psql -U root -d test -h 127.0.0.1 -P
#
brew services stop postgresql@18
brew services start postgresql@18
brew services restart postgresql@18
#
chown -R postgres:postgres /opt/homebrew/var/postgresql@18

Windows - 手工安装 ​

安装目录如下

|---D
    |---Tools
    |---|---PgSQL
    |---|---|---Server                          (安装目录)
    |---|---|---Data                            (数据目录)

安装

bash
# 初始化
.\Server\bin\initdb --pgdata=D:\Tools\PgSQL\Data --encoding=UTF8 --locale=C
# 安装系统服务
.\Server\bin\pg_ctl register --pgdata=D:\Tools\PgSQL\Data -N PgSQL
# 卸载系统服务
.\Server\bin\pg_ctl unregister -N PgSQL
# 启动服务
net start PgSQL
# 停止服务
net stop PgSQL
# 创建测试数据库
.\Server\bin\createdb -E UTF8 test
# 创建用户
.\Server\bin\createuser -d -s -r -P root
.\Server\bin\createuser -d -s -r -P postgres
