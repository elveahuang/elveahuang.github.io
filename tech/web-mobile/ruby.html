# Ruby

# RVM

RVM是一个命令行工具，可以提供一个便捷的多版本 Ruby 环境的管理和切换。

下载地址

# 安装

# 解压缩压缩文件
tar -xzvf rvm-1.29.4.tar.gz -C /Users/elvea/Tools/ruby/
# 进入解压缩后的目录
cd /Users/elvea/Tools/ruby/rvm-1.29.4/
# 删除已有的环境变量
unset GEM_HOME 
# 安装到指定目录 
./install --path /Users/elvea/Tools/ruby/rvm --ignore-dotfiles

# 常用命令

# 列出所有可用版本
rvm list known
# 安装指定版本Ruby
rvm install 2.5.3
# 删除指定版本Ruby
rvm remove 2.5.3
# 查看已安装版本
rvm list
# 查看当前版本
rvm current
# 切换版本
rvm use 2.5.3 --default
# 使用指定版本的Ruby
rvm use default
# 使用系统自带的Ruby
rvm use system

#
echo '[[ -s "$HOME/Tools/ruby/rvm/scripts/rvm" ]] && . "$HOME/Tools/ruby/rvm/scripts/rvm"' >>~/.bashrc
source ~/.bashrc
#
echo '[[ -s "$HOME/Tools/ruby/rvm/scripts/rvm" ]] && . "$HOME/Tools/ruby/rvm/scripts/rvm"' >>~/.zshrc
source ~/.zshrc

# Gem

常用命令

# 切换国内镜像
gem sources --add http://gems.ruby-china.org/
# 切换官方镜像
gem sources --remove https://rubygems.org/
# 列出所有镜像
gem sources -l
# 更新Gem
gem update --system
