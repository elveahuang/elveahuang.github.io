# PHP

# XDebug

# MacOS

# 编译

brew install autoconf

sudo xcode-select -switch /Applications/Xcode.app

/Applications/XAMPP/xamppfiles/bin/phpize-7.2.8

./configure --enable-xdebug --with-php-config=/Applications/XAMPP/xamppfiles/bin/php-config-7.2.8

make

sudo cp modules/* /Applications/XAMPP/xamppfiles/lib/php/extensions/no-debug-non-zts-20170718/

# 安装

[XDebug]
zend_extension = "/Applications/XAMPP/xamppfiles/lib/php/extensions/no-debug-non-zts-20170718/xdebug.so"
xdebug.idekey = "XDebugIDE"
xdebug.remote_enable = 1
xdebug.remote_host = "127.0.0.1"
xdebug.remote_port = 9001

# Windows

[XDebug]
zend_extension = "D:\Dev\php\xampp\php\ext\php_xdebug.dll"
xdebug.idekey = "XDebugIDE"
xdebug.remote_enable = 1
xdebug.remote_handler = "dbgp"
xdebug.remote_host = "127.0.0.1"
xdebug.remote_port = 9001
